fof(thI3,axiom,(
    ! [A,B,C] :
      ( ( subset(A,B)
        & subset(B,C) )
     => subset(A,C) ) )).

fof(set_product,axiom,(
    ! [X] :
      ( set(X)
     => set(product(X)) ) )).

fof(thI42,axiom,(
    ! [A,X] :
      ( member(X,A)
     => subset(product(A),X) ) )).

fof(thV21,conjecture,(
    ! [A] :
      ( ( subset(A,on)
        & set(A)
        & ? [X] : member(X,A) )
     => member(product(A),on) ) )).
