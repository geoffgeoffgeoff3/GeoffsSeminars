    p: $int > $o )).

tff(co1,conjecture,
    ( ( p(0)
      & ! [U: $int] :
          ( p(U)
         => p($sum(U,1)) )
      & ! [V: $int] :
          ( p(V)
         => p($difference(V,1)) ) )
   => ! [W: $int] : p(W) )).
